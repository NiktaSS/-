## ‚úÖ 3. `yield` –∏ –≥–µ–Ω–µ—Ä–∞—Ç–æ—Ä—ã

`yield` —Å–æ–∑–¥–∞—ë—Ç **–ª–µ–Ω–∏–≤—ã–π –∏—Ç–µ—Ä–∞—Ç–æ—Ä**, –∫–æ—Ç–æ—Ä—ã–π –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ –ø–æ –æ–¥–Ω–æ–º—É –∑–∞ —Ä–∞–∑:

```python

# –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä: –∑–∞–¥–∞—á–∞, –∫–æ—Ç–æ—Ä–∞—è —Ä–∞–±–æ—Ç–∞–µ—Ç –≤ 3 —ç—Ç–∞–ø–∞
def task_1():
    print("–ó–∞–¥–∞—á–∞ 1: —Å—Ç–∞—Ä—Ç")
    yield
    print("–ó–∞–¥–∞—á–∞ 1: –ø—Ä–æ–¥–æ–ª–∂–∞–µ–º")
    yield
    print("–ó–∞–¥–∞—á–∞ 1: –∑–∞–≤–µ—Ä—à–µ–Ω–∞")

def task_2():
    print("–ó–∞–¥–∞—á–∞ 2: —Å—Ç–∞—Ä—Ç")
    yield
    print("–ó–∞–¥–∞—á–∞ 2: –ø—Ä–æ–¥–æ–ª–∂–∞–µ–º")
    yield
    print("–ó–∞–¥–∞—á–∞ 2: –∑–∞–≤–µ—Ä—à–µ–Ω–∞")

# –î–æ–±–∞–≤–ª—è–µ–º –∑–∞–¥–∞—á–∏ –≤ –æ—á–µ—Ä–µ–¥—å

tasks.append(task_2())

# –ü—Å–µ–≤–¥–æ-—ç–≤–µ–Ω—Ç-–ª—É–ø: –ø–æ –∫—Ä—É–≥—É –≤—ã–∑—ã–≤–∞–µ–º next() —É –∑–∞–¥–∞—á

    task = tasks.popleft()
    try:
        next(task)  # –≤—ã–ø–æ–ª–Ω—è–µ–º –¥–æ —Å–ª–µ–¥—É—é—â–µ–≥–æ yield
        tasks.append(task)  # —Å—Ç–∞–≤–∏–º –æ–±—Ä–∞—Ç–Ω–æ –≤ –æ—á–µ—Ä–µ–¥—å, –µ—Å–ª–∏ –Ω–µ –∑–∞–≤–µ—Ä—à–∏–ª–∞—Å—å
    except StopIteration:
        pass  # –∑–∞–¥–∞—á–∞ –∑–∞–≤–µ—Ä—à–∏–ª–∞—Å—å, –Ω–µ –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –æ–±—Ä–∞—Ç–Ω–æ
```

üìå –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä—ã —Ç–æ–∂–µ –ø—Ä–∏–æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞—é—Ç—Å—è –∏ –≤–æ–∑–æ–±–Ω–æ–≤–ª—è—é—Ç—Å—è ‚Üí –∏—Ö –º–æ–∂–Ω–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç—å –∫–∞–∫ **–ø—Ä–∏–º–∏—Ç–∏–≤–Ω—É—é –∫–æ—Ä—É—Ç–∏–Ω—É**.
